    // Exponential backoff API fetch helper
    async function fetchWithRetry(url, options, maxRetries = 5) {
      const delays = [1000, 2000, 4000, 8000, 16000];
      for (let i = 0; i < maxRetries; i++) {
        try {
          const response = await fetch(url, options);
          if (response.ok) return await response.json();
          if (response.status === 429 || response.status >= 500) {
            if (i < maxRetries - 1) {
              await new Promise(res => setTimeout(res, delays[i]));
              continue;
            }
          }
          const errData = await response.json().catch(() => ({}));
          throw new Error(errData.error?.message || `เกิดข้อผิดพลาดในการประมวลผล (${response.status})`);
        } catch (err) {
          if (i === maxRetries - 1) throw err;
          await new Promise(res => setTimeout(res, delays[i]));
        }
      }
    }

    // Call Google Gemini API (gemini-2.5-flash-preview-09-2025)
    async function callGeminiApi(systemPrompt, contents, enableSearch = false) {
      const activeKey = getEffectiveApiKey();
      
      if (!activeKey) {
        openApiKeyModal();
        throw new Error("ยังไม่ได้ใส่ Google Gemini API Key ครับ! กรุณากดปุ่ม 🔑 ด้านบนแล้วใส่ API Key ฟรีจาก Google AI Studio ก่อนนะครับ");
      }

      const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${activeKey}`;
      const payload = {
        contents: contents,
        systemInstruction: { parts: [{ text: systemPrompt }] }
      };

      if (enableSearch) {
        payload.tools = [{ "google_search": {} }];
      }

      try {
        const result = await fetchWithRetry(url, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload)
        });
        return result.candidates?.[0]?.content?.parts?.[0]?.text || "ไม่พบคำตอบจาก Google Gemini";
      } catch (err) {
        if (err.message && (err.message.includes('API Key') || err.message.includes('unregistered callers') || err.message.includes('400') || err.message.includes('403'))) {
          openApiKeyModal();
          throw new Error("Google Gemini API Key ไม่ถูกต้องหรือยังไม่ได้ตั้งค่า กรุณาตรวจสอบ API Key ในหน้าต่างที่เปิดขึ้นมาครับ");
        }
        throw err;
      }
    }

    async function extractPdfText(arrayBuffer) {
      if (!window.pdfjsLib) return "ไม่พบ PDF.js library";
      const loadingTask = pdfjsLib.getDocument({ data: arrayBuffer });
      const pdf = await loadingTask.promise;
      let fullText = '';
      for (let i = 1; i <= Math.min(pdf.numPages, 10); i++) {
        const page = await pdf.getPage(i);
        const tokenized = await page.getTextContent();
        const pageText = tokenized.items.map(item => item.str).join(' ');
        fullText += `--- Page ${i} ---\n${pageText}\n\n`;
      }
      return fullText;
    }

    async function handleFireplaceChatSubmit(e) {
      if (e) e.preventDefault();
      const input = document.getElementById('fireplaceChatInput');
      const val = input.value.trim();
      if (!val) return;

      const btn = document.getElementById('btnSendFireplaceChat');
      btn.disabled = true;

      const feed = document.getElementById('fireplaceChatFeed');
      feed.innerHTML += `
        <div class="p-2.5 rounded-2xl bg-orange-500/20 text-orange-100 text-right font-medium border border-orange-500/30">
          <strong>คุณ:</strong> ${val}
        </div>
      `;
      input.value = '';
      feed.scrollTop = feed.scrollHeight;

      try {
        const systemPrompt = `คุณคือ ${mascotState.name} เพื่อนสนิทอบอุ่นริมเตาผิง (Fireplace Companion) ประจำเว็บ CozyDesk AI
ขุมพลังของคุณ: Google Gemini 2.5 Flash + Google Search Grounding

บุคลิกของคุณ: เป็นมิตร อบอุ่น คุยสนุก ชอบฟังเรื่องราว และคุยได้ทุกเรื่องในโลก!

ขอบเขตการคุย:
- คุยได้ทั่วไปทุกเรื่อง ไม่ว่าจะเป็นข่าวดารา เรื่องบันเทิง สภาพอากาศ ดูดวง ไพ่ยิปซี เรื่องเรียน ความรัก หรือการให้กำลังใจฮีลใจ
- ใช้ภาษาไทยแบบเป็นกันเอง อบอุ่น มีคำลงท้ายน่ารักๆ เช่น "ครับ", "น้า", "เพื่อน"
- ใช้เทคโนโลยี Google Search ค้นหาข้อมูลสภาพอากาศ ข่าวดารา หรือข้อมูลปัจจุบันมาตอบผู้ใช้ได้เลย`;

        const contents = [
          { parts: [{ text: `ผู้ใช้ชวนคุยข้างเตาผิงว่า: "${val}"` }] }
        ];

        // Enable Google Search Grounding
        const replyText = await callGeminiApi(systemPrompt, contents, true);

        feed.innerHTML += `
          <div class="p-3 rounded-2xl bg-amber-500/10 border border-amber-500/20 text-amber-200 leading-relaxed space-y-1">
            <div class="font-bold text-amber-300 flex items-center justify-between">
              <span class="flex items-center gap-1.5">
                <span>${mascotState.emoji}</span>
                <span>${mascotState.name}</span>
              </span>
              <span class="text-[8px] bg-blue-500/30 text-blue-200 px-2 py-0.5 rounded-full font-mono">Google Grounded</span>
            </div>
            <p class="whitespace-pre-wrap">${replyText}</p>
          </div>
        `;
      } catch (err) {
        feed.innerHTML += `
          <div class="p-2.5 rounded-2xl bg-rose-950/60 text-rose-300 border border-rose-800">
            ⚠️ ${err.message}
          </div>
        `;
      } finally {
        btn.disabled = false;
        feed.scrollTop = feed.scrollHeight;
      }
    }

    function handleCustomAudioUpload(e) {
      const file = e.target.files[0];
      if (!file) return;
      const player = document.getElementById('localAudioPlayer');
      player.src = URL.createObjectURL(file);
      player.play();
      isPlayingAudio = true;
      document.getElementById('btnMusicPlay').innerText = '⏸️ หยุดเพลง';
      document.getElementById('musicPulse').className = 'w-2 h-2 rounded-full bg-emerald-400 animate-ping';
      showNotification(`🎧 เล่นเพลง: ${file.name}`);
    }

    function renderShopItems() {

    function startFocusTimer() {
      if (isTimerActive) return;
      isTimerActive = true;
      let remaining = totalFocusSeconds;
      focusCountDown = setInterval(() => {
        remaining--;
        let m = Math.floor(remaining / 60);
        let s = remaining % 60;
        document.getElementById('timerClock').innerText = `${m.toString().padStart(2,'0')}:${s.toString().padStart(2,'0')}`;
        if (remaining <= 0) {
          clearInterval(focusCountDown);
          isTimerActive = false;
          window.coins += 50;
          updateStatsUI();
          triggerMemeBreak();
        }
      }, 1000);
    }

    function resetFocusTimer() {
      clearInterval(focusCountDown);
      isTimerActive = false;
      document.getElementById('timerClock').innerText = "25:00";
    }

    function setCustomTimerMinutes() {
      const mins = parseInt(document.getElementById('customMinutesInput').value) || 25;
      totalFocusSeconds = mins * 60;
      document.getElementById('timerClock').innerText = `${mins.toString().padStart(2,'0')}:00`;
    }
